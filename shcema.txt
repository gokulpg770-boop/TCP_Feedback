CREATE TABLE feedback ( id SERIAL PRIMARY KEY, message_id INTEGER NOT NULL REFERENCES messages(id), action VARCHAR(20) NOT NULL, description TEXT, rating INTEGER, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );
